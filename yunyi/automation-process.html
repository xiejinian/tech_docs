02 · P0 最高优先级

内容生产与分发自动化

从投资专家输入核心观点，到各平台内容就绪发布的完整自动化工作流。投资专家全程零代码操作。

⚡
内容生产完整工作流（端到端）
投资专家输入 → AI 生成 → 合规审核 → 平台发布
Step 01
核心观点输入
投资专家
→
飞书多维表格「选题库」新增一行
填写：核心观点（100字内）、内容支柱、目标平台（勾选）、计划发布日期。这是投资专家唯一需要操作的界面。
Step 02
自动触发生成
系统自动
→
飞书触发器 → 调用 Python 脚本 → Claude API
多维表格检测到「状态=待生成」时，自动触发 Webhook → Python 脚本读取观点 → 调用 Claude API → 同步生成5个平台内容草稿 → 写回多维表格对应字段。约 30-60 秒完成。
Step 03
合规审核
谢
→
飞书审批流自动发起 → 谢审核合规措辞
审批单自动创建，包含所有平台草稿内容。谢检查是否有荐股语气、承诺收益等违规表达。可直接在审批单内修改文字。审核耗时目标：<15分钟。
Step 04
内容确认
投资专家
→
投资专家二次确认内容语气和准确性
审批流第二步，投资专家确认 AI 生成的内容是否符合自己的原意表达。可微调措辞，通过后状态自动变为「待发布」。飞书机器人推送发布提醒到群组。
Step 05
平台发布
投资专家 谢
→
自动发布（X）+ 半自动发布（其他平台）
X/Twitter：Python 脚本自动发布 Thread。抖音/小红书/公众号/知识星球：系统生成「一键复制」按钮，投资专家复制对应内容粘贴发布，约 2-3 分钟/平台。发布后在多维表格勾选「已发布」，状态自动归档。

脚本一：飞书触发 → Claude API → 多平台内容生成
核心自动化脚本，部署为 Flask Webhook 服务
content_generator.py
import anthropic
import requests
import json
from flask import Flask, request

app = Flask(__name__)
claude = anthropic.Anthropic(api_key="YOUR_CLAUDE_API_KEY")

FEISHU_APP_ID = "YOUR_APP_ID"
FEISHU_APP_SECRET = "YOUR_APP_SECRET"

SYSTEM_PROMPT = """
你是一位投资知识社群的内容编辑，负责将核心观点改写为各平台适配内容。
核心要求：
1. 所有内容必须以「个人经历分享」口吻，禁止荐股、禁止收益承诺
2. 每条内容末尾附加：「以上为个人经历分享，不构成投资建议。」
3. 输出标准 JSON 格式，包含5个平台的内容字段
"""

def generate_content(core_idea, pillar, background=""):
    user_prompt = f"""
核心观点：{core_idea}
内容支柱：{pillar}
补充背景：{background}

请生成以下平台内容，以 JSON 格式返回：
{{
  "douyin_script": "抖音口播脚本（含强钩子开头，60-180秒）",
  "xiaohongshu": "小红书图文（标题+正文+5个话题标签）",
  "wechat_article": "公众号长文（标题+2000字正文+文末引导）",
  "zsxq_post": "知识星球帖子（500-800字分析+互动问题）",
  "twitter_thread": "X Thread（6条连推，每条<280字，中文为主）"
}}
"""

    message = claude.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=4000,
        system=SYSTEM_PROMPT,
        messages=[{"role": "user", "content": user_prompt}]
    )

    # 解析 JSON 输出
    raw = message.content[0].text
    return json.loads(raw)

def update_feishu_record(record_id, content_dict, table_id):
    # 获取飞书 access_token
    token_resp = requests.post(
        "https://open.feishu.cn/open-apis/auth/v3/tenant_access_token/internal",
        json={"app_id": FEISHU_APP_ID, "app_secret": FEISHU_APP_SECRET}
    )
    token = token_resp.json()["tenant_access_token"]

    # 更新多维表格记录
    requests.put(
        f"https://open.feishu.cn/open-apis/bitable/v1/apps/BASE_ID/tables/{table_id}/records/{record_id}",
        headers={"Authorization": f"Bearer {token}"},
        json={"fields": {
            "抖音脚本": content_dict["douyin_script"],
            "小红书文案": content_dict["xiaohongshu"],
            "公众号草稿": content_dict["wechat_article"],
            "星球帖子": content_dict["zsxq_post"],
            "X Thread": content_dict["twitter_thread"],
            "状态": "待合规审核"
        }}
    )

@app.route("/webhook/generate", methods=["POST"])
def handle_generate():
    data = request.json
    record_id = data["record_id"]
    core_idea = data["core_idea"]
    pillar = data["pillar"]
    background = data.get("background", "")

    content = generate_content(core_idea, pillar, background)
    update_feishu_record(record_id, content, "TOPIC_TABLE_ID")
    return {"status": "ok"}

if __name__ == "__main__":
    app.run(port=5000)

import asyncio
from telegram import Bot
from datetime import datetime, timedelta
import requests

BOT_TOKEN = "YOUR_TELEGRAM_BOT_TOKEN"
CHAT_ID = "YOUR_TELEGRAM_GROUP_ID"  # 电报核心群 ID

def get_expired_members():
    # 从飞书多维表格读取今日到期成员
    token = get_feishu_token()
    today = datetime.now().strftime("%Y-%m-%d")

    resp = requests.get(
        "https://open.feishu.cn/open-apis/bitable/v1/apps/BASE_ID/tables/MEMBER_TABLE/records",
        headers={"Authorization": f"Bearer {token}"},
        params={"filter": f'CurrentValue.[到期时间] <= "{today}" AND CurrentValue.[续费状态] != "已续费"'}
    )
    return resp.json().get("data", {}).get("items", [])

async def kick_expired_members():
    bot = Bot(token=BOT_TOKEN)
    expired = get_expired_members()

    for member in expired:
        tg_id = member["fields"].get("Telegram ID")
        name = member["fields"].get("姓名/昵称")

        if tg_id:
            try:
                # 踢出成员
                await bot.ban_chat_member(
                    chat_id=CHAT_ID,
                    user_id=int(tg_id),
                    until_date=datetime.now() + timedelta(seconds=60)
                )
                # 立即解除封禁（只是踢出不是永久封禁）
                await bot.unban_chat_member(chat_id=CHAT_ID, user_id=int(tg_id))

                update_member_status(member["record_id"], "已到期")

import requests
from datetime import datetime

FEISHU_BOT_WEBHOOK = "https://open.feishu.cn/open-apis/bot/v2/hook/YOUR_WEBHOOK_KEY"

def send_card_message(title, content, color="blue"):
    """发送飞书卡片消息（富文本格式，支持按钮）"""
    payload = {
        "msg_type": "interactive",
        "card": {
            "header": {
                "title": {"tag": "plain_text", "content": title},
                "template": color
            },
            "elements": [{
                "tag": "div",
                "text": {"tag": "lark_md", "content": content}
            }]
        }
    }
    requests.post(FEISHU_BOT_WEBHOOK, json=payload)

def notify_content_approved(topic_id, platform_list):

📋 请前往飞书多维表格复制各平台内容进行发布

def send_daily_task_digest():
    """每日早 8 点推送当日发布任务摘要"""
    today = datetime.now().strftime("%m月%d日")
    # 从飞书多维表格查询今日待发布内容
    tasks = get_today_tasks()

    task_lines = "\n".join([
        f"- **{t['title'][:20]}...** → {' / '.join(t['platforms'])}"
        for t in tasks
    ]) if tasks else "今日暂无排期内容 🎉"

    content = f"""**{today} 内容发布日历**

{task_lines}

📌 X/Twitter 内容将于计划时间自动发布
📌 其他平台请手动复制发布"""
    send_card_message("📅 今日发布任务", content, color="wathet")
